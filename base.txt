{
  config,
  lib,
  pkgs,
  ...
}: let
  cfg = config.services.bore-server;

  # CLI args construction
  args = lib.cli.toGNUCommandLine {} {
    min-port = cfg.minPort;
    max-port = cfg.maxPort;
    bind-addr = cfg.bindAddress;
    bind-tunnels = cfg.bindTunnels;
    secret = null; # Handled via env to avoid logging
  };

  # Wrapper script for ExecStart (to load secret)
  execStartScript = pkgs.writeShellScript "bore-server-wrapper" ''
    #!/usr/bin/env bash
    set -euo pipefail
    ${lib.optionalString (cfg.credentialsFile != null) ''
      export BORE_SECRET=$(cat ''${CREDENTIALS_DIRECTORY}/bore-secret)
    ''}
    exec ${lib.getExe cfg.package} server ${toString args}
  '';
in {
  options.services.bore-server = with lib; {
    enable = mkEnableOption "Bore self-hosted tunnel server";

    package = mkOption {
      type = types.package;
      default = pkgs.bore-cli;
      defaultText = literalExpression "pkgs.bore-cli";
      description = "The bore-cli package to use.";
    };

    bindAddress = mkOption {
      type = types.str;
      default = "0.0.0.0";
      description = "IP address clients must reach the server at (control connection).";
    };

    bindTunnels = mkOption {
      type = types.nullOr types.str;
      default = null;
      description = "IP address for tunnel listeners (defaults to bindAddress if null).";
    };

    minPort = mkOption {
      type = types.ints.between 1 65535;
      default = 1024;
      description = "Minimum TCP port for tunnels.";
    };

    maxPort = mkOption {
      type = types.ints.between 1 65535;
      default = 65535;
      description = "Maximum TCP port for tunnels.";
    };

    credentialsFile = mkOption {
      type = types.nullOr types.path;
      default = null;
      example = "/run/secrets/bore-secret";
      description = ''
        Path to a file containing the authentication secret.
        Use with sops-nix or similar for secure management.
      '';
    };

    openFirewall =
      mkEnableOption "Open firewall ports for control (7835) and tunnel range"
      // {
        default = false;
      };
  };

  config = lib.mkIf cfg.enable {
    networking.firewall = lib.mkIf cfg.openFirewall {
      allowedTCPPorts = [7835]; # Fixed control port
      allowedTCPPortRanges = [
        {
          from = cfg.minPort;
          to = cfg.maxPort;
        }
      ];
    };

    systemd.services.bore-server = {
      description = "Bore Tunnel Server";
      after = ["network.target"];
      wantedBy = ["multi-user.target"];

      serviceConfig = {
        ExecStart = "${execStartScript}";
        Restart = "always";
        RestartSec = "5s";

        DynamicUser = true;
        RuntimeDirectory = "bore-server";
        StateDirectory = "bore-server";

        CapabilityBoundingSet = ["CAP_NET_BIND_SERVICE"];
        AmbientCapabilities = ["CAP_NET_BIND_SERVICE"];
        NoNewPrivileges = true;
        PrivateTmp = true;
        ProtectSystem = "strict";
        ProtectHome = true;
        PrivateDevices = true;
        RestrictAddressFamilies = ["AF_INET" "AF_INET6"];
        RestrictNamespaces = true;
        RestrictSUIDSGID = true;

        LoadCredential = lib.mkIf (cfg.credentialsFile != null) "bore-secret:${cfg.credentialsFile}";
      };
    };
  };

  meta.maintainers = with lib.maintainers; []; # Add your name if upstreaming
}
